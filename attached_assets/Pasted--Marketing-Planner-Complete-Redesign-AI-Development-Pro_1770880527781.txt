```typescript
// Validate Canva link format
function validateCanvaLink(link: string): boolean {
  const regex = /^https:\/\/(www\.)?canva\.com\/design\/[A-Za-z0-9_-]+\/(edit|view)/;
  return regex.test(link);
}

// Extract design ID
function extractCanvaDesignId(link: string): string | null {
  const match = link.match(/canva\.com\/design\/([A-Za-z0-9_-]+)/);
  return match ? match[1] : null;
}

// Main handler
async function handleCanvaLinkSubmission(deliverableId: string, canvaLink: string, userId: string) {
  // 1. Validate link
  if (!validateCanvaLink(canvaLink)) {
    throw new Error('Invalid Canva link format');
  }
  
  const designId = extractCanvaDesignId(canvaLink);
  
  // 2. Update deliverable with link
  await db.update(projectDeliverables)
    .set({
      canva_link: canvaLink,
      canva_design_id: designId,
      design_completed_at: new Date(),
      current_stage: 'COPYWRITING'
    })
    .where(eq(projectDeliverables.id, deliverableId));
  
  // 3. Fetch image from Canva
  await fetchCanvaImage(deliverableId, canvaLink, designId);
  
  // 4. Trigger AI analysis
  await triggerAIAnalysis(deliverableId);
  
  // 5. Send notifications
  const deliverable = await getDeliverableWithProject(deliverableId);
  
  // Notify creator (only if creator != copywriter)
  if (deliverable.project.created_by_id !== deliverable.copywriter_id) {
    await sendSlackNotification({
      userId: deliverable.project.created_by_id,
      type: 'DESIGN_COMPLETED',
      deliverableId,
      message: `Design completed for ${deliverable.deliverable_name}`
    });
  }
  
  // Notify copywriter
  await sendSlackNotification({
    userId: deliverable.copywriter_id,
    type: 'COPY_READY',
    deliverableId,
    message: `Design ready for copy: ${deliverable.deliverable_name}`
  });
}
```

### Step 5: Fetch Image from Canva API

**Endpoint:** `POST /api/deliverables/:id/fetch-canva-image`

```typescript
async function fetchCanvaImage(deliverableId: string, canvaLink: string, designId: string) {
  try {
    // 1. Use Canva REST API to export design as PNG
    const exportResponse = await fetch('https://api.canva.com/rest/v1/exports', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${process.env.CANVA_API_KEY}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        design_id: designId,
        format: 'png',
        pages: [{ page: 1 }]
      })
    });
    
    const exportData = await exportResponse.json();
    const imageUrl = exportData.export.urls[0];
    
    // 2. Download image
    const imageBuffer = await fetch(imageUrl).then(r => r.arrayBuffer());
    
    // 3. Upload to your storage (S3 or local)
    const storedImageUrl = await uploadToStorage(
      imageBuffer,
      `deliverables/${deliverableId}/design.png`
    );
    
    // 4. Generate thumbnail
    const thumbnailUrl = await generateThumbnail(
      imageBuffer,
      `deliverables/${deliverableId}/thumbnail.png`
    );
    
    // 5. Update database
    await db.update(projectDeliverables)
      .set({
        asset_image_url: storedImageUrl,
        asset_thumbnail_url: thumbnailUrl
      })
      .where(eq(projectDeliverables.id, deliverableId));
    
    return { imageUrl: storedImageUrl, thumbnailUrl };
    
  } catch (error) {
    console.error('Canva fetch error:', error);
    // Store error but don't fail - allow manual upload fallback
    await db.update(projectDeliverables)
      .set({ 
        asset_image_url: null,
        asset_thumbnail_url: null 
      })
      .where(eq(projectDeliverables.id, deliverableId));
  }
}
```

```typescript
async function analyzeDesignWithAI(deliverableId: string) {

Based on the image, generate:

Return ONLY valid JSON in this exact format:

    "mood": "...",
    "text_visible": "...",

    "short": "...",
    "medium": "...",
    "long": "..."
  },
  "hashtags": ["#tag1", "#tag2", ...]
}
`;

  const response = await fetch('https://api.anthropic.com/v1/messages', {
```

```typescript
function calculateStageTime(assignedAt: Date | null, completedAt: Date | null): number {
  if (!assignedAt || !completedAt) return 0;
  const ms = completedAt.getTime() - assignedAt.getTime();
  return parseFloat((ms / (1000 * 60 * 60)).toFixed(2)); // Hours
}
```

```typescript
async function logHashtagsToLibrary(
  hashtags: string[],
  brandId: string,

    const normalized = hashtag.toLowerCase().replace('#', '');

        brand_id: brandId,
        region_id: regionId,
        usage_count: 1
      });
```

**Clustering Algorithm:**

```typescript
async function clusterSimilarHashtags(brandId: string) {
  const allTags = await db.query.hashtagLibrary.findMany({
    where: eq(hashtagLibrary.brand_id, brandId)
  });
  
  // Simple similarity: group tags that are 80%+ similar
  const clusters: Map<string, string[]> = new Map();
  
  for (const tag of allTags) {
    let foundCluster = false;
    
    for (const [clusterId, clusterTags] of clusters.entries()) {
      const otherTag = allTags.find(t => t.id === clusterTags[0]);
      if (otherTag && similarity(tag.normalized_tag, otherTag.normalized_tag) > 0.8) {
        clusters.get(clusterId)!.push(tag.id);
        foundCluster = true;
        break;
      }
    }
    
    if (!foundCluster) {
      clusters.set(uuidv4(), [tag.id]);
    }
  }
  
  // Update cluster_id in database
  for (const [clusterId, tagIds] of clusters.entries()) {
    await db.update(hashtagLibrary)
      .set({ cluster_id: clusterId })
      .where(inArray(hashtagLibrary.id, tagIds));
  }
}

function similarity(str1: string, str2: string): number {
  const clean1 = str1.replace(/[_-]/g, '');
  const clean2 = str2.replace(/[_-]/g, '');
  
  if (clean1.includes(clean2) || clean2.includes(clean1)) return 0.9;
  
  // Levenshtein distance calculation
  const matrix: number[][] = [];
  for (let i = 0; i <= str2.length; i++) matrix[i] = [i];
  for (let j = 0; j <= str1.length; j++) matrix[0][j] = j;
  
  for (let i = 1; i <= str2.length; i++) {
    for (let j = 1; j <= str1.length; j++) {
      if (str2[i-1] === str1[j-1]) {
        matrix[i][j] = matrix[i-1][j-1];
      } else {
        matrix[i][j] = Math.min(
          matrix[i-1][j-1] + 1,
          matrix[i][j-1] + 1,
          matrix[i-1][j] + 1
        );
      }
    }
  }
  
  const distance = matrix[str2.length][str1.length];
  const maxLen = Math.max(str1.length, str2.length);
  return 1 - (distance / maxLen);
}
```
