**API Routes:**

// server/routes.ts - Add promotion routes
app.get("/api/promotions", authenticateToken, async (req: any, res) => {
  try {
    const { brandId, status } = req.query;
    const promotions = await storage.getPromotions({ brandId, status });
    res.json(promotions);
  } catch (error) {
    console.error("Get promotions error:", error);
    res.status(500).json({ message: "Internal server error" });
  }
});

app.post("/api/promotions", authenticateToken, async (req: any, res) => {
  try {
    const data = insertPromotionSchema.parse(req.body);
    const promotion = await storage.createPromotion({ 
      ...data, 
      createdById: req.user.userId 
    });
    res.status(201).json(promotion);
  } catch (error) {
    if (error instanceof z.ZodError) {
      return res.status(400).json({ message: "Invalid input", errors: error.errors });
    }
    console.error("Create promotion error:", error);
    res.status(500).json({ message: "Internal server error" });
  }
});

app.put("/api/promotions/:id", authenticateToken, async (req: any, res) => {
  try {
    const promotion = await storage.updatePromotion(req.params.id, req.body);
    if (!promotion) {
      return res.status(404).json({ message: "Promotion not found" });
    }
    res.json(promotion);
  } catch (error) {
    console.error("Update promotion error:", error);
    res.status(500).json({ message: "Internal server error" });
  }
});

```typescript
// client/src/components/unified-calendar.tsx
import { Calendar, momentLocalizer } from 'react-big-calendar';
import moment from 'moment';
import 'react-big-calendar/lib/css/react-big-calendar.css';

const localizer = momentLocalizer(moment);

export function UnifiedCalendar() {
  const [filters, setFilters] = useState({
    brandIds: [],
    platforms: [],
    regionIds: [],
    contentTypes: ['social', 'email', 'event', 'promotion', 'task'],
  });
  
  const { data: brands } = useQuery({ queryKey: ["/api/brands"] });
  const { data: regions } = useQuery({ queryKey: ["/api/regions"] });
  
  // Fetch all content
  const { data: socialPosts } = useQuery({ 
    queryKey: ["/api/social", filters],
    enabled: filters.contentTypes.includes('social'),
  });
  
  const { data: emailCampaigns } = useQuery({ 
    queryKey: ["/api/email", filters],
    enabled: filters.contentTypes.includes('email'),
  });
  
  const { data: events } = useQuery({ 
    queryKey: ["/api/events", filters],
    enabled: filters.contentTypes.includes('event'),
  });
  
  const { data: promotions } = useQuery({ 
    queryKey: ["/api/promotions", filters],
    enabled: filters.contentTypes.includes('promotion'),
  });
  
  const { data: tasks } = useQuery({ 
    queryKey: ["/api/tasks", filters],
    enabled: filters.contentTypes.includes('task'),
  });
  
  // Transform all data into calendar events
  const calendarEvents = useMemo(() => {
    const events = [];
    
    // Social posts
    if (socialPosts) {
      socialPosts
        .filter(applyFilters(filters))
        .forEach(post => {
          const brand = brands?.find(b => b.id === post.brandId);
          events.push({
            id: `social-${post.id}`,
            title: `${post.platform} Post`,
            start: new Date(post.scheduledDate),
            end: new Date(post.scheduledDate),
            resource: { type: 'social', data: post, brand },
            color: brand?.color || '#3b82f6',
          });
        });
    }
    
    // Email campaigns
    if (emailCampaigns) {
      emailCampaigns
        .filter(applyFilters(filters))
        .forEach(email => {
          const brand = brands?.find(b => b.id === email.brandId);
          events.push({
            id: `email-${email.id}`,
            title: email.subject,
            start: new Date(email.scheduledDate),
            end: new Date(email.scheduledDate),
            resource: { type: 'email', data: email, brand },
            color: brand?.color || '#10b981',
          });
        });
    }
    
    // Events
    if (events) {
      events
        .filter(applyFilters(filters))
        .forEach(event => {
          events.push({
            id: `event-${event.id}`,
            title: event.name,
            start: new Date(event.startDate),
            end: new Date(event.endDate),
            resource: { type: 'event', data: event },
            color: '#8b5cf6',
          });
        });
    }
    
    // Promotions
    if (promotions) {
      promotions
        .filter(applyFilters(filters))
        .forEach(promo => {
          const brand = brands?.find(b => b.id === promo.brandId);
          events.push({
            id: `promo-${promo.id}`,
            title: `🎉 ${promo.name}`,
            start: new Date(promo.startDate),
            end: new Date(promo.endDate),
            resource: { type: 'promotion', data: promo, brand },
            color: brand?.color || '#f59e0b',
            allDay: true,
          });
        });
    }
    
    // Tasks
    if (tasks) {
      tasks
        .filter(applyFilters(filters))
        .filter(t => t.dueDate)
        .forEach(task => {
          const brand = brands?.find(b => b.id === task.brandId);
          events.push({
            id: `task-${task.id}`,
            title: `📋 ${task.title}`,
            start: new Date(task.dueDate),
            end: new Date(task.dueDate),
            resource: { type: 'task', data: task, brand },
            color: brand?.color || '#6b7280',
          });
        });
    }
    
    return events;
  }, [socialPosts, emailCampaigns, events, promotions, tasks, filters, brands]);
  
  const eventStyleGetter = (event) => {
    return {
      style: {
        backgroundColor: event.color,
        borderColor: event.color,
        color: 'white',
      }
    };
  };
  
  return (
    <div>
      {/* Filters */}
      <Card className="mb-4">
        <CardContent className="pt-6">
          <div className="grid grid-cols-1 md:grid-cols-4 gap-4">
            {/* Brand Filter */}
            <MultiSelect
              label="Brands"
              options={brands?.map(b => ({ label: b.name, value: b.id })) || []}
              value={filters.brandIds}
              onChange={(value) => setFilters({...filters, brandIds: value})}
            />
            
            {/* Platform Filter */}
            <MultiSelect
              label="Platforms"
              options={[
                { label: 'Instagram', value: 'INSTAGRAM' },
                { label: 'Facebook', value: 'FACEBOOK' },
                { label: 'LinkedIn', value: 'LINKEDIN' },
                { label: 'Twitter', value: 'TWITTER' },
                { label: 'TikTok', value: 'TIKTOK' },
                { label: 'Email', value: 'EMAIL' },
              ]}
              value={filters.platforms}
              onChange={(value) => setFilters({...filters, platforms: value})}
            />
            
            {/* Region Filter */}
            <MultiSelect
              label="Regions"
              options={regions?.map(r => ({ label: r.name, value: r.id })) || []}
              value={filters.regionIds}
              onChange={(value) => setFilters({...filters, regionIds: value})}
            />
            
            {/* Content Type Filter */}
            <MultiSelect
              label="Content Types"
              options={[
                { label: 'Social Posts', value: 'social' },
                { label: 'Email Campaigns', value: 'email' },
                { label: 'Events', value: 'event' },
                { label: 'Promotions', value: 'promotion' },
                { label: 'Tasks', value: 'task' },
              ]}
              value={filters.contentTypes}
              onChange={(value) => setFilters({...filters, contentTypes: value})}
            />
          </div>
          
          <div className="flex justify-end mt-4">
            <Button 
              variant="ghost" 
              size="sm"
              onClick={() => setFilters({
                brandIds: [],
                platforms: [],
                regionIds: [],
                contentTypes: ['social', 'email', 'event', 'promotion', 'task'],
              })}
            >
              Reset Filters
            </Button>
          </div>
        </CardContent>
      </Card>
      
      {/* Calendar */}
      <Card>
        <CardContent className="pt-6">
          <Calendar
            localizer={localizer}
            events={calendarEvents}
            startAccessor="start"
            endAccessor="end"
            style={{ height: 700 }}
            eventPropGetter={eventStyleGetter}
            onSelectEvent={(event) => handleEventClick(event)}
            views={['month', 'week', 'day', 'agenda']}
            defaultView="month"
          />
        </CardContent>
      </Card>
    </div>
  );
}

function applyFilters(filters) {
  return (item) => {
    // Brand filter
    if (filters.brandIds.length > 0 && item.brandId) {
      if (!filters.brandIds.includes(item.brandId)) return false;
    }
    
    // Platform filter
    if (filters.platforms.length > 0 && item.platform) {
      if (!filters.platforms.includes(item.platform)) return false;
    }
    
    // Region filter
    if (filters.regionIds.length > 0 && item.regionId) {
      if (!filters.regionIds.includes(item.regionId)) return false;
    }
    
    return true;
  };
}
```

// client/src/components/recurrence-selector.tsx
export function RecurrenceSelector({ value, onChange }) {
  const [recurrence, setRecurrence] = useState(value || {
    enabled: false,
    frequency: 'WEEKLY',
    interval: 1,
    daysOfWeek: [],
    endDate: null,
    occurrences: null,
  });
  
  return (
    <div className="space-y-4">
      <div className="flex items-center space-x-2">
        <Switch
          checked={recurrence.enabled}
          onCheckedChange={(checked) => {
            const updated = { ...recurrence, enabled: checked };
            setRecurrence(updated);
            onChange(updated);
          }}
        />
        <Label>Repeat this content</Label>
      </div>
      
      {recurrence.enabled && (
        <>
          {/* Frequency */}
          <div className="space-y-2">
            <Label>Repeat every</Label>
            <div className="flex gap-2">
              <Input
                type="number"
                min="1"
                value={recurrence.interval}
                onChange={(e) => {
                  const updated = { ...recurrence, interval: parseInt(e.target.value) };
                  setRecurrence(updated);
                  onChange(updated);
                }}
                className="w-20"
              />
              <Select
                value={recurrence.frequency}
                onValueChange={(value) => {
                  const updated = { ...recurrence, frequency: value };
                  setRecurrence(updated);
                  onChange(updated);
                }}
              >
                <SelectTrigger className="w-32">
                  <SelectValue />
                </SelectTrigger>
                <SelectContent>
                  <SelectItem value="DAILY">Day(s)</SelectItem>
                  <SelectItem value="WEEKLY">Week(s)</SelectItem>
                  <SelectItem value="MONTHLY">Month(s)</SelectItem>
                  <SelectItem value="YEARLY">Year(s)</SelectItem>
                </SelectContent>
              </Select>
            </div>
          </div>
          
          {/* Days of week for weekly recurrence */}
          {recurrence.frequency === 'WEEKLY' && (
            <div className="space-y-2">
              <Label>Repeat on</Label>
              <div className="flex flex-wrap gap-2">
                {['MONDAY', 'TUESDAY', 'WEDNESDAY', 'THURSDAY', 'FRIDAY', 'SATURDAY', 'SUNDAY'].map(day => (
                  <Button
                    key={day}
                    variant={recurrence.daysOfWeek?.includes(day) ? "default" : "outline"}
                    size="sm"
                    onClick={() => {
                      const days = recurrence.daysOfWeek || [];
                      const updated = {
                        ...recurrence,
                        daysOfWeek: days.includes(day)
                          ? days.filter(d => d !== day)
                          : [...days, day]
                      };
                      setRecurrence(updated);
                      onChange(updated);
                    }}
                  >
                    {day.slice(0, 3)}
                  </Button>
                ))}
              </div>
            </div>
          )}
          
          {/* End condition */}
          <div className="space-y-2">
            <Label>Ends</Label>
            <RadioGroup
              value={recurrence.endDate ? 'date' : recurrence.occurrences ? 'after' : 'never'}
              onValueChange={(value) => {
                if (value === 'never') {
                  const updated = { ...recurrence, endDate: null, occurrences: null };
                  setRecurrence(updated);
                  onChange(updated);
                }
              }}
            >
              <div className="flex items-center space-x-2">
                <RadioGroupItem value="never" id="never" />
                <Label htmlFor="never">Never</Label>
              </div>
              
              <div className="flex items-center space-x-2">
                <RadioGroupItem value="date" id="date" />
                <Label htmlFor="date">On</Label>
                <Input
                  type="date"
                  value={recurrence.endDate || ''}
                  onChange={(e) => {
                    const updated = { ...recurrence, endDate: e.target.value, occurrences: null };
                    setRecurrence(updated);

// server/recurrenceService.ts
import { addDays, addWeeks, addMonths, addYears, isBefore, isAfter } from 'date-fns';

export class RecurrenceService {
  static generateOccurrences(
    baseDate: Date,
    pattern: RecurrencePattern,
    maxDate?: Date
  ): Date[] {
    const occurrences: Date[] = [baseDate];
    let currentDate = new Date(baseDate);
    let count = 1;
    
    const endDate = pattern.endDate ? new Date(pattern.endDate) : maxDate;
    const maxOccurrences = pattern.occurrences || 100; // Safety limit
    
    while (count < maxOccurrences) {
      // Calculate next date based on frequency
      switch (pattern.frequency) {
        case 'DAILY':
          currentDate = addDays(currentDate, pattern.interval);
          break;
        case 'WEEKLY':
          currentDate = addWeeks(currentDate, pattern.interval);
          // Adjust to specified days of week
          if (pattern.daysOfWeek && pattern.daysOfWeek.length > 0) {
            // Implementation for specific days
          }
          break;
        case 'MONTHLY':
          currentDate = addMonths(currentDate, pattern.interval);
          break;
        case 'YEARLY':
          currentDate = addYears(currentDate, pattern.interval);
          break;
      }
      
      // Check if we've exceeded the end date
      if (endDate && isAfter(currentDate, endDate)) {
        break;
      }
      
      occurrences.push(new Date(currentDate));
      count++;
    }
    
    return occurrences;
  }
  
  static async createRecurringInstances(
    parentId: string,
    parentType: 'social' | 'email' | 'event',
    pattern: RecurrencePattern,
    storage: IStorage
  ) {
    let parent;
    switch (parentType) {
      case 'social':
        parent = await storage.getSocialPost(parentId);
        break;
      case 'email':
        parent = await storage.getEmailCampaign(parentId);
        break;
      case 'event':
        parent = await storage.getEvent(parentId);
        break;
    }
    
    if (!parent) throw new Error('Parent not found');
    
    // Generate future dates
    const baseDate = new Date(parent.scheduledDate || parent.startDate);
    const occurrences = this.generateOccurrences(baseDate, pattern);
    
    // Create instances (skip first as it's the parent)
    for (let i = 1; i < occurrences.length; i++) {
      const date = occurrences[i];
      
      switch (parentType) {
        case 'social':
          await storage.createSocialPost({
            ...parent,
            id: undefined,
            scheduledDate: date,
            recurrenceParentId: parentId,
            isRecurrenceParent: false,
          });
          break;
        case 'email':
          await storage.createEmailCampaign({
            ...parent,
            id: undefined,
            scheduledDate: date,
            recurrenceParentId: parentId,
            isRecurrenceParent: false,
          });
          break;
        case 'event':
          await storage.createEvent({
            ...parent,
            id: undefined,
            startDate: date,
            endDate: addDays(date, differenceInDays(parent.endDate, parent.startDate)),
            recurrenceParentId: parentId,
            isRecurrenceParent: false,
          });
          break;
      }
    }
  }
}
