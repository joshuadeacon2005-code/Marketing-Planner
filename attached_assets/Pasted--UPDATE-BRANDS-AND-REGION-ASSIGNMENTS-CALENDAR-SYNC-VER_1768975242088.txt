## STEP 1: DATABASE SCHEMA UPDATE

### 1.1 Update Brand-Region Relationship

**Current Schema Issue:**
Brands table has a single `regionId` field, but brands can operate in MULTIPLE regions.

**Solution:**
Create a many-to-many relationship table.

**New Schema:**
```typescript
// In shared/schema.ts

// Remove regionId from brands table (if it exists)
export const brands = pgTable("brands", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  name: text("name").notNull().unique(),
  color: text("color").notNull().default("#F7971C"),
  isActive: boolean("is_active").notNull().default(true),
  dreamPimUrl: text("dream_pim_url"),
  assetPortalUrl: text("asset_portal_url"),
  assetPortalName: text("asset_portal_name").default("Asset Portal"),
  description: text("description"),
  createdAt: timestamp("created_at").notNull().defaultNow(),
});

// NEW: Brand-Region junction table
export const brandRegions = pgTable("brand_regions", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  brandId: varchar("brand_id").notNull().references(() => brands.id, { onDelete: "cascade" }),
  regionId: varchar("region_id").notNull().references(() => regions.id, { onDelete: "cascade" }),
  createdAt: timestamp("created_at").notNull().defaultNow(),
}, (table) => ({
  // Ensure unique brand-region combinations
  uniqueBrandRegion: unique().on(table.brandId, table.regionId),
}));

// Relations
export const brandsRelations = relations(brands, ({ many }) => ({
  regions: many(brandRegions),
}));

export const brandRegionsRelations = relations(brandRegions, ({ one }) => ({
  brand: one(brands, { fields: [brandRegions.brandId], references: [brands.id] }),
  region: one(regions, { fields: [brandRegions.regionId], references: [regions.id] }),
}));

// Types
export const insertBrandRegionSchema = createInsertSchema(brandRegions);
export const selectBrandRegionSchema = createSelectSchema(brandRegions);
export type BrandRegion = z.infer<typeof selectBrandRegionSchema>;
export type InsertBrandRegion = z.infer<typeof insertBrandRegionSchema>;
```

  try {
    // Get all regions
    const allRegions = await db.select().from(regions);
    const regionMap = Object.fromEntries(
      allRegions.map(r => [r.code, r.id])

## STEP 3: UPDATE STORAGE LAYER

### 3.1 Add Brand-Region Query Methods

**File: `server/storage.ts`**

```typescript
export interface IStorage {
  // ... existing methods ...
  
  // NEW: Brand-Region methods
  getBrandRegions(brandId: string): Promise<BrandRegion[]>;
  getRegionBrands(regionId: string): Promise<Brand[]>;
  assignBrandToRegion(brandId: string, regionId: string): Promise<BrandRegion>;
  removeBrandFromRegion(brandId: string, regionId: string): Promise<boolean>;
}

export class DatabaseStorage implements IStorage {
  // ... existing implementations ...
  
  // NEW: Get all regions for a brand
  async getBrandRegions(brandId: string): Promise<BrandRegion[]> {
    return await db
      .select()
      .from(brandRegions)
      .where(eq(brandRegions.brandId, brandId));
  }
  
  // NEW: Get all brands available in a region
  async getRegionBrands(regionId: string): Promise<Brand[]> {
    const brandRegionRecords = await db
      .select({
        brand: brands,
      })
      .from(brandRegions)
      .innerJoin(brands, eq(brandRegions.brandId, brands.id))
      .where(eq(brandRegions.regionId, regionId));
    
    return brandRegionRecords.map(record => record.brand);
  }
  
  // NEW: Assign brand to region
  async assignBrandToRegion(brandId: string, regionId: string): Promise<BrandRegion> {
    const [created] = await db
      .insert(brandRegions)
      .values({ brandId, regionId })
      .returning();
    return created;
  }
  
  // NEW: Remove brand from region
  async removeBrandFromRegion(brandId: string, regionId: string): Promise<boolean> {
    await db
      .delete(brandRegions)
      .where(
        and(
          eq(brandRegions.brandId, brandId),
          eq(brandRegions.regionId, regionId)
        )
      );
    return true;
  }
  
  // UPDATE: Get brand with its regions
  async getBrand(id: string): Promise<Brand & { regions?: Region[] } | undefined> {
    const [brand] = await db.select().from(brands).where(eq(brands.id, id));
    if (!brand) return undefined;
    
    // Get associated regions
    const brandRegionRecords = await db
      .select({
        region: regions,
      })
      .from(brandRegions)
      .innerJoin(regions, eq(brandRegions.regionId, regions.id))
      .where(eq(brandRegions.brandId, id));
    
    return {
      ...brand,
      regions: brandRegionRecords.map(record => record.region),
    };
  }
  
  // UPDATE: Get all brands with their regions
  async getBrands(): Promise<Array<Brand & { regions?: Region[] }>> {
    const allBrands = await db.select().from(brands).orderBy(asc(brands.name));
    
    // Get all brand-region relationships
    const allBrandRegions = await db
      .select()
      .from(brandRegions)
      .innerJoin(regions, eq(brandRegions.regionId, regions.id));
    
    // Map regions to brands
    return allBrands.map(brand => ({
      ...brand,
      regions: allBrandRegions
        .filter(br => br.brand_regions.brandId === brand.id)
        .map(br => br.regions),
    }));
  }
}
```

// Get regions for a specific brand
app.get("/api/brands/:id/regions", authenticateToken, async (req: any, res) => {
  try {
    const brandRegions = await storage.getBrandRegions(req.params.id);
    res.json(brandRegions);
  } catch (error) {
    console.error("Get brand regions error:", error);
    res.status(500).json({ message: "Internal server error" });
  }
});

// Get brands available in a specific region
app.get("/api/regions/:id/brands", authenticateToken, async (req: any, res) => {
  try {
    const brands = await storage.getRegionBrands(req.params.id);
    res.json(brands);
  } catch (error) {
    console.error("Get region brands error:", error);
    res.status(500).json({ message: "Internal server error" });
  }
});

// Assign brand to region (admin only)
app.post("/api/brands/:brandId/regions/:regionId", authenticateToken, requireAdmin, async (req: any, res) => {
  try {
    const { brandId, regionId } = req.params;
    const brandRegion = await storage.assignBrandToRegion(brandId, regionId);
    res.status(201).json(brandRegion);
  } catch (error) {
    console.error("Assign brand to region error:", error);
    res.status(500).json({ message: "Internal server error" });
  }
});

// Remove brand from region (admin only)
app.delete("/api/brands/:brandId/regions/:regionId", authenticateToken, requireAdmin, async (req: any, res) => {
  try {
    const { brandId, regionId } = req.params;
    await storage.removeBrandFromRegion(brandId, regionId);
    res.status(204).send();
  } catch (error) {
    console.error("Remove brand from region error:", error);
    res.status(500).json({ message: "Internal server error" });
  }
});

           ↓
           ↓ [Query Layer]

    ┌──────┴───────┬────────────┬────────────┐

Calendar       Calendar     Calendar     Calendar

```

  b.name as brand_name,
  r.code as region_code,
  r.name as region_name
FROM brand_regions br
JOIN brands b ON br.brand_id = b.id
JOIN regions r ON br.region_id = r.id
ORDER BY b.name, r.code;

-- Count brands per region
SELECT 
  r.code,
  r.name,
  COUNT(br.brand_id) as brand_count
FROM regions r
LEFT JOIN brand_regions br ON r.id = br.region_id
GROUP BY r.code, r.name
ORDER BY r.code;
```
